  #!/bin/bash
  oc secrets new docker-pull-secret .dockerconfigjson=${HOME}/.docker/config.json --namespace=\$1 2&gt;/dev/null

    echo "Created secret in namespace \$1" && sleep 3		
    oc secrets link default docker-pull-secret --for=pull --namespace=\$1		
  fi
